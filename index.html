<script>

function showToast(m){const t=document.getElementById('toast');t.textContent=m;t.classList.add('show');clearTimeout(t._timeout);t._timeout=setTimeout(()=>t.classList.remove('show'),3000);}

const agentes={'GC-12345':{nombre:'Carlos García',pass:'1234'},'GC-67890':{nombre:'Ana Martínez',pass:'5678'}};

let agenteActual=null,fotos=[],mediciones=[],transcripcion='',hechos='',inspeccion='',numDil=0;

let localidadActual='Madrid', delitoActual='Robo con fuerza';

let firmaD={puntos:[],firmado:false},firmaA={puntos:[],firmado:false},ctxD,ctxA;


function login(){const t=document.getElementById('tipInput').value.trim();const p=document.getElementById('passInput').value.trim();if(agentes[t]&&agentes[t].pass===p){agenteActual={tip:t,nombre:agentes[t].nombre};document.getElementById('loginOverlay').classList.remove('active');document.getElementById('appMain').style.display='block';document.getElementById('agenteNombre').textContent=agenteActual.nombre;document.getElementById('agenteTip').textContent='TIP: '+t;showToast('✅ Bienvenido');inicializarFirmas();}else{showToast('❌ TIP o contraseña incorrectos');}}

document.getElementById('btnLogin').addEventListener('click',login);

document.getElementById('passInput').addEventListener('keypress',e=>{if(e.key==='Enter')login();});

document.getElementById('btnLogout').addEventListener('click',function(){agenteActual=null;document.getElementById('appMain').style.display='none';document.getElementById('loginOverlay').classList.add('active');showToast('👋 Sesión cerrada');});


function inicializarFirmas(){

    const cD=document.getElementById('canvasDenunciante');

    if(cD){

        cD.width=cD.parentElement.offsetWidth||300;cD.height=120;ctxD=cD.getContext('2d');ctxD.strokeStyle='#1a6a9a';ctxD.lineWidth=2.5;ctxD.lineCap='round';

        cD.addEventListener('mousedown',e=>{const r=cD.getBoundingClientRect();firmaD.puntos=[{x:(e.clientX-r.left)*(cD.width/r.width),y:(e.clientY-r.top)*(cD.height/r.height)}];dibujarFirma(ctxD,firmaD.puntos);});

        cD.addEventListener('mousemove',e=>{if(firmaD.puntos.length===0)return;const r=cD.getBoundingClientRect();firmaD.puntos.push({x:(e.clientX-r.left)*(cD.width/r.width),y:(e.clientY-r.top)*(cD.height/r.height)});dibujarFirma(ctxD,firmaD.puntos);});

        cD.addEventListener('mouseup',()=>{if(firmaD.puntos.length>5){firmaD.firmado=true;document.getElementById('firmaDenuncianteArea').classList.add('firmado');showToast('✅ Firma denunciante');}});

        cD.addEventListener('touchstart',e=>{e.preventDefault();const t=e.touches[0];const r=cD.getBoundingClientRect();firmaD.puntos=[{x:(t.clientX-r.left)*(cD.width/r.width),y:(t.clientY-r.top)*(cD.height/r.height)}];dibujarFirma(ctxD,firmaD.puntos);},{passive:false});

        cD.addEventListener('touchmove',e=>{e.preventDefault();if(firmaD.puntos.length===0)return;const t=e.touches[0];const r=cD.getBoundingClientRect();firmaD.puntos.push({x:(t.clientX-r.left)*(cD.width/r.width),y:(t.clientY-r.top)*(cD.height/r.height)});dibujarFirma(ctxD,firmaD.puntos);},{passive:false});

        cD.addEventListener('touchend',e=>{e.preventDefault();if(firmaD.puntos.length>5){firmaD.firmado=true;document.getElementById('firmaDenuncianteArea').classList.add('firmado');showToast('✅ Firma denunciante');}},{passive:false});

    }

    const cA=document.getElementById('canvasAgente');

    if(cA){

        cA.width=cA.parentElement.offsetWidth||300;cA.height=120;ctxA=cA.getContext('2d');ctxA.strokeStyle='#2d7d46';ctxA.lineWidth=2.5;ctxA.lineCap='round';

        cA.addEventListener('mousedown',e=>{const r=cA.getBoundingClientRect();firmaA.puntos=[{x:(e.clientX-r.left)*(cA.width/r.width),y:(e.clientY-r.top)*(cA.height/r.height)}];dibujarFirma(ctxA,firmaA.puntos);});

        cA.addEventListener('mousemove',e=>{if(firmaA.puntos.length===0)return;const r=cA.getBoundingClientRect();firmaA.puntos.push({x:(e.clientX-r.left)*(cA.width/r.width),y:(e.clientY-r.top)*(cA.height/r.height)});dibujarFirma(ctxA,firmaA.puntos);});

        cA.addEventListener('mouseup',()=>{if(firmaA.puntos.length>5){firmaA.firmado=true;document.getElementById('firmaAgenteArea').classList.add('firmado');showToast('✅ Firma agente');}});

        cA.addEventListener('touchstart',e=>{e.preventDefault();const t=e.touches[0];const r=cA.getBoundingClientRect();firmaA.puntos=[{x:(t.clientX-r.left)*(cA.width/r.width),y:(t.clientY-r.top)*(cA.height/r.height)}];dibujarFirma(ctxA,firmaA.puntos);},{passive:false});

        cA.addEventListener('touchmove',e=>{e.preventDefault();if(firmaA.puntos.length===0)return;const t=e.touches[0];const r=cA.getBoundingClientRect();firmaA.puntos.push({x:(t.clientX-r.left)*(cA.width/r.width),y:(t.clientY-r.top)*(cA.height/r.height)});dibujarFirma(ctxA,firmaA.puntos);},{passive:false});

        cA.addEventListener('touchend',e=>{e.preventDefault();if(firmaA.puntos.length>5){firmaA.firmado=true;document.getElementById('firmaAgenteArea').classList.add('firmado');showToast('✅ Firma agente');}},{passive:false});

    }

}


function dibujarFirma(ctx,p){ctx.clearRect(0,0,ctx.canvas.width,ctx.canvas.height);if(p.length<2)return;ctx.beginPath();ctx.moveTo(p[0].x,p[0].y);for(let i=1;i<p.length;i++)ctx.lineTo(p[i].x,p[i].y);ctx.stroke();}

function limpiarFirmas(){firmaD={puntos:[],firmado:false};firmaA={puntos:[],firmado:false};if(ctxD)ctxD.clearRect(0,0,ctxD.canvas.width,ctxD.canvas.height);if(ctxA)ctxA.clearRect(0,0,ctxA.canvas.width,ctxA.canvas.height);document.getElementById('firmaDenuncianteArea').classList.remove('firmado');document.getElementById('firmaAgenteArea').classList.remove('firmado');showToast('🧹 Firmas limpiadas');}

window.limpiarFirmas=limpiarFirmas;


document.getElementById('btnDeclaracion').addEventListener('click',function(){if(!agenteActual){showToast('⚠️ Inicia sesión');return;}

const loc=prompt('📍 Indique la Localidad de los hechos:', localidadActual);if(loc) localidadActual=loc;const del=prompt('📁 Indique el Delito presunto:', delitoActual);if(del) delitoActual=del;const dec=prompt('🎤 Introduzca la declaración del denunciante:');if(dec){transcripcion=dec;hechos=dec;document.getElementById('diligenciaBody').textContent='🔄 Declaración grabada. Pulsa EJECUTAR.';document.getElementById('exposicionBody').textContent='📝 '+transcripcion;showToast('✅ Declaración grabada');}});

document.getElementById('btnInspeccion').addEventListener('click',function(){if(!agenteActual){showToast('⚠️ Inicia sesión');return;}const insp=prompt('👁️ Describa el resultado de la Inspección Ocular:');if(insp){inspeccion=insp;document.getElementById('inspeccionBody').textContent=inspeccion;showToast('✅ Inspección registrada');}});

document.getElementById('btnVideo').addEventListener('click',function(){showToast('🎥 Vídeo grabado (simulado)');});

document.getElementById('btnFoto').addEventListener('click',function(){const d=prompt('📸 Descripción de la fotografía tomada:');if(d){fotos.push(d);showToast('📸 Foto añadida');}});

document.getElementById('btnMedicion').addEventListener('click',function(){const d=prompt('📏 Ángulo / Medición efectuada:');if(d){mediciones.push(d);showToast('📏 Medición guardada');}});


document.getElementById('btnEjecutar').addEventListener('click',function(){if(!transcripcion){showToast('⚠️ Graba primero la exposición');return;}if(!firmaD.firmado||!firmaA.firmado){showToast('⚠️ Firmas obligatorias en pestaña Firm');return;}numDil++;const n=String(numDil).padStart(6,'0');const f=new Date().toLocaleDateString();const h=new Date().toLocaleTimeString();const a=agenteActual.tip;

const d=`DILIGENCIA Nº ${n}\n\n${localidadActual}, ${f} ${h}\nAGENTE: ${a}\n\n"${transcripcion}"\n\nFdo.: Denunciante / ${a}`;

const e=`EXPOSICIÓN Nº ${n}\n\n${localidadActual}, ${f}\n${hechos}\n\nFdo.: Denunciante / ${a}`;

const i=`INSPECCIÓN Nº ${n}\n\n${inspeccion||'No realizada.'}\nFOTOS: ${fotos.length}\nMEDICIONES: ${mediciones.length}`;

const c=`CARÁTULA Nº ${n}\n\nFECHA: ${f}\nAGENTE: ${a}\nDELITO: ${delitoActual}\nREMITIDO: Juzgado de ${localidadActual}`;

const o=`OFICIO JUZGADO ${localidadActual.toUpperCase()}\n\n${localidadActual}, ${f}\n\nCOMANDANTE`;

document.getElementById('diligenciaBody').textContent=d;document.getElementById('exposicionBody').textContent=e;document.getElementById('inspeccionBody').textContent=i;document.getElementById('caratulaBody').textContent=c;document.getElementById('oficioBody').textContent=o;showToast('✅ Diligencias generadas');});


document.querySelectorAll('.tab-btn').forEach(b=>{b.addEventListener('click',function(){document.querySelectorAll('.tab-btn').forEach(x=>x.classList.remove('active'));this.classList.add('active');document.querySelectorAll('.doc-section').forEach(x=>x.classList.remove('active'));document.getElementById('tab-'+this.dataset.tab).classList.add('active');if(this.dataset.tab==='firmas')setTimeout(inicializarFirmas,120);});});

document.getElementById('btnExportar').addEventListener('click',function(){const c=`=== DILIGENCIA ===\n${document.getElementById('diligenciaBody').textContent}\n\n=== EXPOSICIÓN ===\n${document.getElementById('exposicionBody').textContent}\n\n=== INSPECCIÓN ===\n${document.getElementById('inspeccionBody').textContent}\n\n=== CARÁTULA ===\n${document.getElementById('caratulaBody').textContent}\n\n=== OFICIO ===\n${document.getElementById('oficioBody').textContent}`;const b=new Blob([c],{type:'text/plain'});const l=document.createElement('a');l.href=URL.createObjectURL(b);l.download='Diligencias.txt';l.click();showToast('📄 Exportado');});

document.getElementById('btnLimpiar').addEventListener('click',function(){if(confirm('¿Limpiar todo?')){document.querySelectorAll('.doc-body').forEach(el=>el.textContent='--');document.getElementById('diligenciaBody').textContent='Pulse "Exposición" para comenzar.';transcripcion='';hechos='';inspeccion='';fotos=[];mediciones=[];limpiarFirmas();showToast('🧹 Todo Limpiado');}});
